#include <pthread.h>
#include <stdio.h>
#include <stdlib.h>
#include <stdint.h>  // for intptr_t

#define NUM_THREADS 3

int je = 0, jo = 0;
int evensum = 0, sumn = 0, oddsum = 0;
int evenarr[50], oddarr[50];

void *Even(void *threadid)
{
    int i, n;
    je = 0;
    n = (int)(intptr_t)threadid;

    for (i = 1; i <= n; i++)
    {
        if (i % 2 == 0)
        {
            evenarr[je] = i;
            evensum += i;
            je++;
        }
    }

    pthread_exit(NULL);
}

void *Odd(void *threadid)
{
    int i, n;
    jo = 0;
    n = (int)(intptr_t)threadid;

    for (i = 1; i <= n; i++)
    {
        if (i % 2 != 0)
        {
            oddarr[jo] = i;
            oddsum += i;
            jo++;
        }
    }

    pthread_exit(NULL);
}

void *SumN(void *threadid)
{
    int i, n;
    n = (int)(intptr_t)threadid;

    for (i = 1; i <= n; i++)
    {
        sumn += i;
    }

    pthread_exit(NULL);
}

int main()
{
    pthread_t threads[NUM_THREADS];
    int t, i;

    printf("Enter a number: ");
    scanf("%d", &t);

    // Creating threads
    pthread_create(&threads[0], NULL, Even, (void *)(intptr_t)t);
    pthread_create(&threads[1], NULL, Odd, (void *)(intptr_t)t);
    pthread_create(&threads[2], NULL, SumN, (void *)(intptr_t)t);

    // Waiting for threads to finish
    for (i = 0; i < NUM_THREADS; i++)
    {
        pthread_join(threads[i], NULL);
    }

    // Output results
    printf("\nThe sum of first %d natural numbers is: %d\n", t, sumn);
    printf("The sum of even numbers up to %d is: %d\n", t, evensum);
    printf("The sum of odd numbers up to %d is: %d\n", t, oddsum);

    printf("\nEven numbers up to %d:\n", t);
    for (i = 0; i < je; i++)
    {
        printf("%d ", evenarr[i]);
    }

    printf("\n\nOdd numbers up to %d:\n", t);
    for (i = 0; i < jo; i++)
    {
        printf("%d ", oddarr[i]);
    }

    printf("\n");

    pthread_exit(NULL);
}
